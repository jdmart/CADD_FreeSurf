266  FUNCTION zbqlbin(N,P)
267 *
268 * Returns a random number binomially distributed (N,P)
269 *
270  DOUBLE PRECISION P,ZBQLBET1
271  DOUBLE PRECISION PP,PPP,G,Y,TINY
272  INTEGER N,ZBQLBIN,ZBQLGEO,IZ,NN
273 
274  tiny = 1.0d-8

276 
277  IF (.NOT.( (p.GE.0.0d0).AND.(p.LE.1.0d0) )) THEN
278  WRITE(*,1)
279  RETURN
280  ELSEIF(n.LE.0) THEN
281  WRITE(*,1)
282  RETURN
283  ENDIF
284 *
285 * First step: if NP > 10, say, things will be expensive, and
286 * we can get into the right ballpark by guessing a value for
287 * ZBQLBIN (IZ, say), and simulating Y from a Beta distribution
288 * with parameters IZ and NN-IZ+1 (NN starts off equal to N).
289 * If Y is less than PP (which starts off as P) then the IZth order
290 * statistic from NN U(0,1) variates is less than P, and we know
291 * that there are at least IZ successes. In this case we focus on
292 * the remaining (NN-IZ) order statistics and count how many are
293 * less than PP, which is binomial (NN-IZ,(PP-Y)/(1-Y)).
294 * Otherwise, if Y is greater than PP there must be less
295 * than IZ successes, so we can count the number of order statistics
296 * under PP, which is binomial (IZ-1,P/Y). When we've got NN*PP
297 * small enough, we go to the next stage of the algorithm and
298 * generate the final bits directly.
299 *
300  nn = n
301  pp = p
302  10 iz = int(dble(nn)*pp) + 1
303  IF ( (iz.GT.10).AND.(iz.LT.nn-10) ) THEN
304  y = zbqlbet1(dble(iz),dble(nn-iz+1))
305  IF (y.LT.pp) THEN
306  zbqlbin = zbqlbin + iz
307  nn = nn - iz
308  pp = (pp-y) / (1.0d0-y)
309  ELSE
310  nn = iz-1
311  pp = pp/y
312  ENDIF
313  GOTO 10
314  ENDIF
315 *
316 * PP is the probability of the binomial we're currently
317 * simulating from. For the final part, we simulate either number
318 * of failures or number of success, depending which is cheaper.
319 *
320  20 IF (pp.GT.0.5) THEN
321  ppp = 1.0d0-pp
322  ELSE
323  ppp = pp
324  ENDIF
325 
326  g = 0
327  iz = 0
328 *
329 * ZBQLGEO falls over for miniscule values of PPP, so ignore these
330 * (tiny probability of any successes in this case, anyway)
331 *
332  IF (ppp.GT.tiny) THEN
333  30 g = g + zbqlgeo(ppp)
334  IF (g.LE.nn) THEN
335  iz = iz + 1
336  GOTO 30
337  ENDIF
338  ENDIF
339 
340  IF (pp.GT.0.5) iz = nn - iz
341  zbqlbin = zbqlbin + iz
342 
343  1 FORMAT(/5x,'****ERROR**** Illegal parameter value in ',
344  +' ZBQLBIN',/)
345  END

385  FUNCTION zbqlpoi(MU)
386 *
387 * Returns a random number Poisson distributed with mean MU
388 *
389 
390  DOUBLE PRECISION ZBQLU01,X,Y,MU,PI
391  DOUBLE PRECISION ZBQLLG,ZBQLGAM,MU1,TMP1,TMP2,T
392  INTEGER ZBQLPOI,ZBQLBIN,K,INIT
393  SAVE init,pi
394  DATA init /0/
395 
396  IF (init.EQ.0) THEN
397  pi = 4.0d0*datan(1.0d0)
398  init = 1
399  ENDIF

401  zbqlpoi = 0
402 
403  IF (mu.LT.0.0d0) THEN
404  WRITE(*,1)
405  RETURN
406  ENDIF

408 * For small MU, generate exponentials till their sum exceeds 1
409 * (equivalently, uniforms till their product falls below e^-MU)
410 *
411  IF (mu.LE.1.0d3) THEN
412  mu1 = mu
413 *
414 * For values of MU less than 1000, use order statistics - the Kth
415 * event in a Poisson process of rate MU has a Gamma distribution
416 * with parameters (MU,K); if it's greater than 1 we know that there
417 * are less than K events in (0,1) (and the exact number is binomial)
418 * and otherwise the remaining number is another Poisson. Choose K so
419 * that we'll get pretty close to 1 in the first go but are unlikely
420 * to overshoot it.
421 *
422  19 IF (mu1.GT.1.0d1) THEN
423  k = int(mu1-dsqrt(mu1))
424  y = zbqlgam(dble(k),mu1)
425  IF (y.GT.1.0d0) THEN
426  zbqlpoi = zbqlpoi + zbqlbin(k-1,(1.0d0/y))
427  RETURN
428  ENDIF
429  zbqlpoi = zbqlpoi + k
430  mu1 = mu1 * (1.0d0-y)
431  GOTO 19
432  ENDIF
433  y = dexp(-mu1)
434  x = 1.0d0
435  20 x = x*zbqlu01(0.0d0)
436  IF (x.GT.y) THEN
437  zbqlpoi = zbqlpoi + 1
438  GOTO 20
439  ENDIF
440 *
441 * For really huge values of MU, use rejection sampling as in
442 * Press et al (1992) - large numbers mean some accuracy may be
443 * lost, but it caps the execution time.
444 *
445  ELSE
446  tmp1 = dsqrt(2.0d0*mu)
447  tmp2 = zbqllg(mu+1.0d0)-(mu*dlog(mu))
448  30 y = dtan(pi*zbqlu01(0.0d0))
449  zbqlpoi = int(mu + (tmp1*y))

544  FUNCTION zbqlbet1(NU1,NU2)
545 *
546 * Returns a random number, beta distributed with degrees
547 * of freedom NU1 and NU2.
548 *
549  DOUBLE PRECISION NU1,NU2,ZBQLGAM,ZBQLBET1,ZBQLU01,X1,X2
550 
551  zbqlbet1 = 0.0d0
552 
553  IF ( (nu1.LE.0.0).OR.(nu2.LE.0.0) ) THEN
554  WRITE(*,1)
555  RETURN
556  ENDIF
557 *
558 * If parameters are too small, gamma subroutine tends to return zero
559 * as all the probability goes to the origin and we get rounding
560 * errors, even with double precision. In this case, we use Johnk's
561 * method, suitably scaled to avoid rounding errors as much as possible.
562 *
563 
564  IF ( (nu1.LT.0.9d0).AND.(nu2.LT.0.9d0) ) THEN
565  10 x1 = zbqlu01(0.0d0)
566  x2 = zbqlu01(0.0d0)
567  IF ( (x1**(1.0d0/nu1))+(x2**(1.0d0/nu2)).GT.1.0d0) GOTO 10
568  x1 = (dlog(x2)/nu2) - (dlog(x1)/nu1)
569  zbqlbet1 = (1.0d0 + dexp(x1))**(-1)
570  IF (zbqlbet1.GT.1.0d0) GOTO 10
571  ELSE
572  x1 = zbqlgam(nu1,1.0d0)
573  x2 = zbqlgam(nu2,1.0d0)
574  zbqlbet1 = x1/(x1+x2)
575  ENDIF
576 
577  RETURN
578

580  +' ZBQLBET1',/5x,

659  FUNCTION zbqllg(X)
660 *
661 * Returns log(G(X)) where G is the Gamma function. The algorithm is
662 * that given in Press et al (1992), Section 6.1, although this
663 * version also allows for arguments less than 1.
664 *

668  DATA init /0/
669  DATA (c(i),i=0,6) /
670  + 1.000000000190015d0,76.18009172947146d0,
671  + -86.50532032941677d0,24.01409824083091d0,
672  + -1.231739572450155d0,0.1208650973866179d-2,

678  init = 1
679  ENDIF

682 * our working argument.

697  tmp = ( (z-0.5d0)*dlog(tmp) ) - tmp + rln2p

699  sum = c(0)
700  DO 50 i=1,6
701  sum = sum + (c(i)/(z+dble(i-1)))
702  50 CONTINUE

709  zbqllg = dlog(tmp/dsin(tmp)) - zbqllg

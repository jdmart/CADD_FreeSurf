99  read *,nmaterials
100  print*, 'number of materials', nmaterials
101  if (nmaterials.lt.1) then
102  print *,'***ERROR: Illegal number of materials specified.'
103  stop
104  endif
105  allocate(material(nmaterials))
106  nbasismax=0
107  do i=1,nmaterials
108  if(key.eq.'dire') then
109  iunit=5
110  else
111  read (*,'(a)') datafile
112  if (.not.fileexists(datafile,.true.)) stop
113  call iofile(datafile,'formatted ',iunit,.true.)
114  endif
115  read(iunit,*) material(i)%name
116  read(iunit,*) (material(i)%bvec(:,j),j=1,3)
117  read(iunit,*) material(i)%nbasis
118  allocate(material(i)%basis(3,material(i)%nbasis))
119  material(i)%basis = 0.
120  allocate(material(i)%ispec(material(i)%nbasis))
121  material(i)%ispec = 0
122  read(iunit,*) (material(i)%basis(:,j),material(i)%ispec(j),j=1
123  $ ,material(i)%nbasis)
124  read(iunit,'(a3)') material(i)%structure
125  read(iunit,*) material(i)%a0
126  read(iunit,*) ncc
127  material(i)%cc=0.d0
128  do j=1,ncc
129  read(iunit,*) icc,jcc,ctmp
130  material(i)%cc(icc,jcc)=ctmp
131  material(i)%cc(jcc,icc)=ctmp
132  enddo
133  do j=1,3
134  material(i)%bmag(j) = dsqrt(dot_product(material(i)%bvec(:
135  $ ,j),material(i)%bvec(:,j)))
136  enddo
137  material(i)%volume = det33(material(i)%bvec)
138  if(material(i)%volume.le.0.d0) then
139  write(*,*) '** ERROR: unit cell volume is negative
140  $ or zero for material',i
141  write(*,*) 'Make sure Bravais lattice is right-handed'
142  stop
143  endif
144  if (nbasismax.lt.material(i)%nbasis) nbasismax =
145  $ material(i)%nbasis
146  if(key.ne.'dire') close(iunit)
147  enddo
148 
149  error=.false.
150  if(ndf.lt.3*nbasismax) then
151  write(*,*) '***ERROR: ndf is too small to accomodate the'
152  write(*,*) ' defined materials.'
153  write(*,*) ' Increase ndf to:',3*nbasismax
154  error=.true.
155  endif
156  if(error) stop
157  return
158  end subroutine readmaterials

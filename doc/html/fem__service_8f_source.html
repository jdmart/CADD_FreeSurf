143  function fe_area(lmn)
144  implicit none
145  include 'fem_parameters.par'

148 c
149  double precision xl(knode*ndof)
150  double precision pn(knode), qn(knode)
151  double precision xjac(ndof, ndof), det, area
152  integer i, j
153 c
154 c Everything is hard-coded, since I don't know where I may need
155 c the derivatives of shape functions
156 c
157 c Shape functions are: N_1 = 1 - p - q; N_2 = p; N_3 = q;
158 c
159  pn(1)=-1.0d0
160  pn(2)=1.0d0
161  pn(3)=0.0d0
162  qn(1)=-1.0d0
163  qn(2)=0.0d0
164  qn(3)=1.0d0
165 c
166  call fe_find(lmn, xl)
167  do 2 i=1,ndof
168  do 2 j=1,ndof
169 2 xjac(i,j)=0.0d0
170 c
171  do i=1, knode
172  xjac(1,1)=xjac(1,1)+pn(i)*xl((i-1)*ndof+1)
173  xjac(1,2)=xjac(1,2)+qn(i)*xl((i-1)*ndof+1)
174  xjac(2,1)=xjac(2,1)+pn(i)*xl((i-1)*ndof+2)
175  xjac(2,2)=xjac(2,2)+qn(i)*xl((i-1)*ndof+2)
176  enddo
177 c
178  det = xjac(1,1)*xjac(2,2) - xjac(1,2)*xjac(2,1)
179  fe_area = abs(det)/2.0d0
180 c
181  return
182  end

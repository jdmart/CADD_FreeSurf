37 !------------------------------------------------------------------
38 ! BuildCluster -- Build a cluster of atoms
39 !
40 ! Passed Parameters :
41 ! mat (in) : type(bravaismat), contains the material
42 ! from which to build the cluster
43 ! rcut (in) : radius of cluste
44 ! atoms(out): type(cluster), contains the built cluster
45 !
46 ! Algorithm :
47 ! Span the directions of the three Bravais vectors,
48 ! but try to be clever about minimizing the range of
49 ! each loop (important for non-orthogonal bravais lattice
50 ! vectors).
51 !
52 ! Add to the cluster all atoms associated with bravais
53 ! sites less than rcut from the origin, making the origin
54 ! site atoms first in the list.
55 !
56 ! Notes :
57 ! Needs to be rigourously tested for speed and optimized
58 ! if possible - this will be called for every call to
59 ! "cauchyborn"
60 !
61 ! Author :
62 ! R. Miller (01/17/98)
63 !
64 ! Revisions :
65 ! none
66 !
67 !--
68  subroutine buildcluster(mat,rcut,atoms)
69  use mod_material
70  implicit none
71 
72 ! variables that are passed
73  type(bravaismat) mat
74  double precision rcut
75  type(cluster) atoms

77 !-- Local Variables
78 !
79  type(cluster) at
80  integer ii(3),i1,i2,i3,is1,is2,is3,if1,if2,if3,ic1,ic2,ic3,nguess
81  $ ,i,j,ia,ib,ib2
82  double precision, parameter :: PI43=4.18879d0
83  double precision latlen(3),org(2),per3(2),bperp,rlen2,r(3),rb(3)
84  $ ,per2,cperp,rcut2,vec(3),rtol
85  logical notzero1,notzero2,notzero3
86 !
87 
88  rcut2 = rcut*rcut
89 !
90 ! estimate cluster size using (volume of sphere)/(volume per atom),
91 ! with
92 ! a safty factor of 2
93 !
94  nguess=2*nint(mat%nbasis*(pi43*rcut*rcut2)/(mat%volume))
95  allocate (at%x(3,nguess))
96  allocate (at%rlat(mat%nbasis,nguess))
97  allocate (at%spec(nguess))
98  allocate (at%lattice(nguess))
99 ! C@
100  at%x = 0.
101  at%rlat = 0.
102  at%spec = 0
103  at%lattice = 0
104 !
105 ! compute lengths of bravais lattice vectors
106 !
107  1 do j=1,3
108  latlen(j)=dot_product(mat%bvec(1:3,j),mat%bvec(1:3,j))
109  latlen(j)=sqrt(latlen(j))
110  enddo
111 !
112 ! to optimize speed, sort (using sorting map ii) the bvec.
113 ! Call the bvec vectors "a" "b" and "c". "a" is chosen to be
114 ! the shortest of the three vectors. Of the remaining 2, "b" is
115 ! the one that is "least perpendicular" to "a". Finally, "c" is the
116 ! remaining vector. map these using ii(1)=a, ii(2)=b, ii(3)=c.
117 !
118 ! bperp is the length of b perp. to a. Similarly cperp.
119 ! "per3" is the projection of c in the plane defined by a and b
120 ! "per2" is the projection of b along a.
121 !
122  call sortbravais(mat%bvec,latlen,ii,bperp,cperp,per3,per2)
123 !
124 ! Put in the center site atoms (bravais lattice (0,0,0))
125 ! now, so that they appear first in list.
126 !
127  at%natoms=0
128  do ib=1,mat%nbasis
129  do j=1,3
130  rb(j)= mat%basis(ii(1),ib)*mat%bvec(j,ii(1))
131  $ + mat%basis(ii(2),ib)*mat%bvec(j,ii(2))
132  $ + mat%basis(ii(3),ib)*mat%bvec(j,ii(3))
133  enddo
134  at%natoms=at%natoms+1
135  at%x(1:3,at%natoms)=rb
136  at%spec(at%natoms)=mat%ispec(ib)
137  at%lattice(at%natoms)=ib
138  enddo
139 ! update the rlat
140  do ia=1,at%natoms
141  do ib=1,at%natoms
142  if(ia.eq.ib) then
143  at%rlat(ib,ia)=0
144  else
145  vec=at%x(1:3,ia)-at%x(1:3,ib)
146  at%rlat(ib,ia)=sqrt(dot_product(vec,vec))
147  endif
148  enddo
149  enddo
150 
151 !
152 ! decide how big to make the loop over the "c" direction
153 !
154  if3=int(rcut/cperp)
155  is3=int(-rcut/cperp)
156  if (if3.gt.is3) then
157  ic3=1
158  else
159  ic3=-1
160  endif
161 !
162 ! loop in the c direction
163 !
164  do i3=is3,if3,ic3
165  notzero3=i3.ne.0
166  org(2)=i3*per3(2)
167 !
168 ! decide how big to make the loop over the "b" direction
169 !
170  if2=int((rcut-org(2))/bperp)
171  is2=int((-rcut-org(2))/bperp)
172  if (if2.gt.is2) then
173  ic2=1
174  else
175  ic2=-1
176  endif
177 !
178 ! loop in the b direction
179 !
180  do i2=is2,if2,ic2
181  notzero2=i2.ne.0
182  org(1)=i3*per3(1)+i2*per2
183 !
184 ! decide how big to make the loop over the "a" direction
185 !
186  if1=int((rcut-org(1))/latlen(ii(1)))
187  is1=int((-rcut-org(1))/latlen(ii(1)))
188  if (if1.gt.is1) then
189  ic1=1
190  else
191  ic1=-1
192  endif
193 !
194 ! loop in the a direction
195 !
196  do i1=is1,if1,ic1
197  notzero1=i1.ne.0
198  if(notzero1.or.notzero2.or.notzero3) then
199  rlen2=0.
200  do j=1,3
201  r(j) = i1*mat%bvec(j,ii(1))
202  $ + i2*mat%bvec(j,ii(2))
203  $ + i3*mat%bvec(j,ii(3))
204  rlen2 = rlen2 + r(j)*r(j)
205  enddo
206  if(rlen2.le.rcut2) then
207 !
208 ! add all atoms associated with this BL site
209 !
210  do ib=1,mat%nbasis
211  do j=1,3
212  rb(j) = r(j) + mat%basis(ii(1),ib)*mat%bvec(j
213  $ ,ii(1))
214  $ + mat%basis(ii(2),ib)*mat%bvec(j,ii(2))
215  $ + mat%basis(ii(3),ib)*mat%bvec(j,ii(3))
216  enddo
217 ! add an atom
218  at%natoms=at%natoms+1
219 !
220 ! make sure the assumed array size is still okay. If not, start
221 ! again.
222 !
223  if(at%natoms.gt.nguess) then
224  write(*,*)
225  $ '***WARNING: BuildCluster had to reallocate!'
226  nguess=nguess*2
227  allocate (at%x(3,nguess))
228  allocate (at%rlat(mat%nbasis,nguess))
229  allocate (at%spec(nguess))
230  allocate (at%lattice(nguess))
231 !C@
232  at%x = 0.
233  at%rlat = 0.
234  at%spec = 0
235  at%lattice = 0
236 
237  go to 1
238  endif
239  at%x(1:3,at%natoms)=rb
240  at%spec(at%natoms)=mat%ispec(ib)
241  at%lattice(at%natoms)=ib
242 ! compute rlat
243  do ib2=1,mat%nbasis
244  vec=at%x(1:3,at%natoms)-at%x(1:3,ib2)
245  at%rlat(ib2,at%natoms)=sqrt(dot_product(vec
246  $ ,vec))
247  enddo
248  enddo
249  endif
250  endif
251  enddo
252  enddo
253  enddo
254 !
255 ! Now allocate the array to be returned, and copy the data from
256 ! the temporary storage
257 !
258  atoms%natoms=at%natoms
259  allocate (atoms%x(3,at%natoms))
260 !C@
261  atoms%x = 0.
262  atoms%x(1:3,1:at%natoms)=at%x(1:3,1:at%natoms)
263  allocate (atoms%rlat(mat%nbasis,at%natoms))
264  atoms%rlat = 0.
265  atoms%rlat(1:mat%nbasis,1:at%natoms)=at%rlat(1:mat%nbasis
266  $ ,1:at%natoms)
267  allocate (atoms%spec(at%natoms))
268  atoms%spec = 0
269  atoms%spec(1:at%natoms)=at%spec(1:at%natoms)
270  allocate (atoms%lattice(at%natoms))
271  atoms%lattice = 0
272  atoms%lattice(1:at%natoms)=at%lattice(1:at%natoms)
273 !
274 ! deallocate the temporary storage
275 !
276  call deallocate_cluster(at)
277  end subroutine buildcluster

307  ! SortBravais -- Pre-processing for buildcluster
308  !
309  ! Passed Parameters :
310  ! lat (in): bravais lattice vectors
311  ! latlen(in): lengths of lat
312  ! ii (out): sort map for lat a=1,b=2,c=3
313  ! bperp(out): component of b perp. to a
314  ! bperp(out): component of c perp. to a
315  ! per3 (out): projection of c on a-b plane
316  ! per2 (out): projection of b on a
317  !
318  ! Algorithm :
319  ! make the shortest vector "a", make "b" the one
320  ! that is least perpendicular to "a" and "c" is the leftover
321  !
322  ! Notes :
323  ! Needs to be rigourously tested for speed and optimized
324  ! if possible - this will be called for every call to "cauchyborn"
325  !
326  ! Author :
327  ! R. Miller (01/17/98)
328  !
329  ! Revisions :
330  ! none
331  !
332  !--

334  subroutine sortbravais(lat,latlen,ii,bperp,cperp,per3,per2)
335  implicit none
336  integer ii(3),j,i,itemp
337  double precision lat(3,3),bperp,per3(2),latlen(3),t3(3),per2,cperp
338  $ ,t2(3),bperp2,bperp3,temp2,temp3,per22,per23
339 !
340 ! make the shortest vector "a"
341 !
342  ii(1)=1
343  do i=2,3
344  if(latlen(i).lt.latlen(ii(1))) ii(1)=i
345  enddo
346  ii(2)=mod(ii(1),3)+1
347  ii(3)=mod(ii(2),3)+1
348 !
349 ! bperp is the length of the component of b perp. to a.
350 ! bperp2 is using vector 2 as b,
351 ! bperp3 is using vector 3 as b.
352 !
353 ! choose the shorter of bperp2,bperp3 to be bperp
354 !
355  per22=dot_product(lat(1:3,ii(1)),lat(1:3,ii(2)))/latlen(ii(1))
356  temp2=per22/latlen(ii(1))
357  per23=dot_product(lat(1:3,ii(1)),lat(1:3,ii(3)))/latlen(ii(1))
358  temp3=per23/latlen(ii(1))
359  bperp2=0.
360  bperp3=0.
361  do j=1,3
362  t2(j)=lat(j,ii(2))-temp2*lat(j,ii(1))
363  bperp2=bperp2+t2(j)**2
364  t3(j)=lat(j,ii(3))-temp3*lat(j,ii(1))
365  bperp3=bperp3+t3(j)**2
366  enddo
367  if(bperp2.lt.bperp3) then
368  bperp=dsqrt(bperp2)
369  per2=per22
370  per3(1)=dot_product(lat(1:3,ii(3)),lat(1:3,ii(1)))
371  $ /latlen(ii(1))
372  per3(2)=dot_product(lat(1:3,ii(3)),t2)/bperp
373  else
374  bperp=dsqrt(bperp3)
375  per2=per23
376  itemp=ii(2)
377  ii(2)=ii(3)
378  ii(3)=itemp
379  per3(1)=dot_product(lat(1:3,ii(3)),lat(1:3,ii(1)))
380  $ /latlen(ii(1))
381  per3(2)=dot_product(lat(1:3,ii(3)),t3)/bperp
382  endif
383 !
384  call cross_product(lat(1:3,ii(1)),lat(1:3,ii(2)),t3)
385  cperp=dot_product(t3,lat(1:3,ii(3)))/sqrt(dot_product(t3,t3))
386  end subroutine sortbravais
